How to collect data on a CVM

chmod +x perf-collect.sh
./perf-collect.sh --cluster-id my-cluster-01

# Optional parameters:
./perf-collect.sh --cluster-id my-cluster-01 \
    --duration 30 \
    --frequency 99 \
    --pid 12345
